Save as CssSelectors.html /* Simple selectors */ /* Element selector */ h1 { color: blue; } /* ID selector */ #container { padding: 20px; border: 1px solid black; } /* Class selector */ .title { font-size: 2em; text-align: center; } /* Group selector */ p, li { font-family: Arial, sans-serif; } /* Universal selector */ * { margin: 0; padding: 0; } /* Combinator selectors */ /* Descendant selector */ div ul { margin: 20px 0; } /* Child selector */ div > ul { list-style-type: none; } /* Adjacent sibling selector */ h1 + p { color: grey; } /* General sibling selector */ p ~ ul { color: green; } /* Pseudo-class selectors */ li:first-child { font-weight: bold; } li:nth-child(2) { color: red; } button:hover { background-color: yellow; } /* Pseudo-element selectors */ p::before { content: "Note: "; font-weight: bold; color: red; } p::after { content: " Thank you for reading."; color: green; } /* Attribute selectors */ input[type="text"] { border: 1px solid blue; } input[placeholder="Type here"] { color: grey; }
